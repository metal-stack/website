import { useState, useEffect } from "react";

export function VersionedLink({ children }) {
  const [version, setVersion] = useState("");

  useEffect(() => {
    const path = window.location.pathname;
    const v = path
      .replace("/docs/", "")
      .replace("references/api", "")
      .replace("/", "");
    if (v === "") setVersion("latest");
    else setVersion(v);
  }, []);

  if (version) return <a href={`/api?version=${version}`}>{children}</a>;

  return null;
}
